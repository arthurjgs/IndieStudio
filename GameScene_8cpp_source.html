292 {
293  Type::Vector<2> playerPos = { playerPosition.getX(), playerPosition.getZ() };
294  Type::Vector<2> cubicMap = this->_gameMap.lock()->getCubicMap();
295  int cubicMapX = static_cast<int>(cubicMap.getX());
296  int cubicMapY = static_cast<int>(cubicMap.getY());
297  float playerRadius = 0.3f;
298  std::vector<Type::Color> mapPixels = _gameMap.lock()->getMapPixels();
299 
300  int playerCellX = static_cast<int>(round(playerPos.getX() - _gameMap.lock()->getPosition().getX() + 0.5f));
301  int playerCellY = static_cast<int>(round(playerPos.getY() - _gameMap.lock()->getPosition().getZ() + 0.5f));
302 
303  if (playerCellX < 0) playerCellX = 0;
304  else if (playerCellX >= cubicMapX) playerCellX = cubicMapX - 1;
305 
306  if (playerCellY < 0) playerCellY = 0;
307  else if (playerCellY >= cubicMapY) playerCellY = cubicMapY - 1;
308 
309  for (int y = 0; y < cubicMapY; y++)
310  {
311  for (int x = 0; x < cubicMapX; x++)
312  {
313  if ((mapPixels[y * cubicMapX + x].getR() == 255) &&
314  (RayLib::Models::Collision::CheckCollisionCircleRec(playerPos, playerRadius,
315  Type::Rectangle { _gameMap.lock()->getPosition().getX() - 0.5f + x * 1.0f,
316  _gameMap.lock()->getPosition().getZ() - 0.5f + y * 1.0f,
317  1.0f,
318  1.0f })))
319  {
320  return true;
321  }
322  }
323  }

544 void Bomberman::GameScene::update(const double &elapsed)
545 {
546  if (this->__lastAlive()) {
547  RayLib::Window::loadingScreen();
548  this->__manager.clearStack<End>();
549  return;
550  }
551  std::vector<int> sideList;
552  this->_everySecond += elapsed;
553  this->_timerCamera += elapsed;
554  float &x = _camera.getPosition().getX();
555  float &z = _camera.getPosition().getZ();
556 
557  if (_checkCamera)
558  {
559  int random = rand() % 2;
560  int randomSign = rand() % 2;
561  int randomNumber = rand() % 3 + 1;
562  if (random == 1)
563  {
564  if (randomSign == 0)
565  x += randomNumber;
566  else
567  x -= randomNumber;
568  }
569  else if (random == 0)
570  {
571  if (randomSign == 0)
572  z += randomNumber;
573  else
574  z -= randomNumber;
575  }
576  }
577  if (_timerCamera > 1)
578  {
579  _checkCamera = false;
580  _timerCamera = 0.0;
581  x = _cameraOriginX;
582  z = _cameraOriginZ;
583  }
584 
585  // CHECK IF PAUSE HIS ON GOING
586  this->updatePause(elapsed);
587  if (this->_pause == true || this->quitting) {
588  return;
589  }
590 
591  // CHECK IF BOMB HAS EXPLODED
592  for (auto & b : _bombList) {
593  if (b.expired())
594  continue;
595  if (b.lock()->getState() == GameObject::DESTROYED) {
596  _soundExplosion.PlaySound();
597  auto flames = b.lock()->explode();
598  for (auto & flame : flames) {
599  if (std::find(sideList.begin(), sideList.end(), flame->getSide()) != sideList.end())
600  continue;
601  if (checkCollisionForObjects(flame->getPosition(), true) == BASIC ||
602  checkCollisionForMap(flame->getPosition()))
603  sideList.emplace_back(flame->getSide());
604  _gameObjectList.emplace_back(flame);
605  _soundFlame.PlaySound();
606  }
607  for (auto &val : this->_listPlayers)
608  {
609  val.lock()->updateCollisions(b.lock()->getPosition(), 0);
610  val.lock()->updateDangers(b.lock()->getPosition(), b.lock()->getRange(), 0);
611  }
612  }
613  }
614 
615 
616  // CHECK IF OBJECTS SHOULD BE DESTROYED
617  _gameObjectList.erase(std::remove_if(
618  _gameObjectList.begin(), _gameObjectList.end(),
619  [] (std::shared_ptr<Bomberman::GameObject> &b) {
620  return b->getState() == GameObject::DESTROYED;
621  }), _gameObjectList.end());
622 
623  for (auto & object : _gameObjectList) {
624  if (object->getType() != GameObject::PLAYER)
625  object->update(elapsed);
626  }
627 
628  for (auto & player : _listPlayers) {
629  if (!player.lock()->getAlive()) {
630  if (this->_everySecond > 1.0) {
631  player.lock()->setDisplay(false);
632  }
633  continue;
634  }
635  if (player.lock()->getState() == Player::PlayerState::ACTION) {
636  double angle = player.lock()->getRotationAngle();
637  auto position = player.lock()->getPosition();
638  auto bombPos = Type::Vector<3>();
639 
640  if (angle == 0)
641  bombPos = Type::Vector<3>(static_cast<float>(round(position.getX())),
642  static_cast<float>(round(position.getY())),
643  static_cast<float>(round(position.getZ() + 1)));
644  else if (angle == 180)
645  bombPos = Type::Vector<3>(static_cast<float>(round(position.getX())),
646  static_cast<float>(round(position.getY())),
647  static_cast<float>(round(position.getZ() - 1)));
648  else if (angle == 90)
649  bombPos = Type::Vector<3>(static_cast<float>(round(position.getX() + 1)),
650  static_cast<float>(round(position.getY())),
651  static_cast<float>(round(position.getZ())));
652  else if (angle == -90)
653  bombPos = Type::Vector<3>(static_cast<float>(round(position.getX()) - 1),
654  static_cast<float>(round(position.getY())),
655  static_cast<float>(round(position.getZ())));
656  if (checkCollisionForMap(bombPos) || checkCollisionForObjects(bombPos, false, true) == BASIC) {
657  player.lock()->setState(Player::PlayerState::IDLE);
658  continue;
659  }
660  std::shared_ptr<Bomb> bomb = player.lock()->createBomb();
661  _soundBombFuse.PlaySound();
662  if (bomb != nullptr) {
663 
664  for (auto &val : this->_listPlayers)
665  {
666  val.lock()->updateCollisions(bomb->getPosition(), 1);
667  val.lock()->updateDangers(bomb->getPosition(), bomb->getRange(), 1);
668  }
669  _gameObjectList.emplace_back(bomb);
670  _bombList.emplace_back(bomb);
671  }
672  }
673  auto oldPosition = player.lock()->getPosition();
674  player.lock()->update(elapsed);
675  if (checkCollisionForMap(player.lock()->getPosition()))
676  player.lock()->setPosition(oldPosition);
677  COLLIDE_EVENT ret = checkCollisionForObjects(player);
678  if (ret == BASIC)
679  player.lock()->setPosition(oldPosition);
680  if (ret == DEATH) {
681  _checkCamera = true;
682  // A PLAYER DEATH START HERE
683  player.lock()->setAlive(false);
684  player.lock()->setState(Player::PlayerState::DEAD);
685  player.lock()->setPosition(oldPosition);
686  this->_everySecond = 0.0;
687  }
688  }
689  this->_second += elapsed;
690  if (this->_second >= 1) {
691  this->_timer--;
692  std::string &text = this->getTextFromName("timer").lock()->getText();
693  text = this->convertSecondToDisplayTime(this->_timer);
694  if (this->_timer == 0) {
695  RayLib::Window::loadingScreen();
696  this->__manager.clearStack<End>();
697  return;
698  }
699  this->_second = 0.0;
700  }
701  for (auto const &val : this->_2DGameObjectList) {
702  val.second->update(elapsed);
703  }
704  for (auto const &val : this->_2DButtonList) {
705  if (val.second.lock()->isClick() && val.second.lock()->getDisplay()) {
706  if (this->_buttonCallback.count(val.second.lock()->getName()) > 0) {
707  this->_buttonCallback[val.second.lock()->getName()](*this);
708  }
709  }
710  }
